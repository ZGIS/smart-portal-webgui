      localStorage.removeItem('currentUser');
      localStorage.removeItem('currentUserProfile');
      this.cookieService.remove('XSRF-TOKEN');
    }
    // TODO SR add "you've been logged out" to message
    return this.handleError(errorResponse);
  }
